CREATE TABLE public.employee
(
	id					SERIAL PRIMARY KEY NOT NULL,
    user_id				INT,
    first_name 			CHAR(16) NOT NULL,
    last_name 			CHAR(32) NOT NULL,
    phone_number 		CHAR(9),
    mobile_phone_number	CHAR(9),
    employment_date		DATE NOT NULL,
    job					CHAR(16) NOT NULL,
    sex					CHAR(1) NOT NULL,
    pesel				CHAR(11) UNIQUE NOT NULL,
    birthday_date		DATE NOT NULL,
    residence_address	CHAR(64) NOT NULL,
    mailing_address		CHAR(64),
    email				CHAR(32),
    salary				DECIMAL(9,2),
    bonus				DECIMAL(9,2)   
);

ALTER TABLE public.employee
ADD CONSTRAINT fk_user_id
FOREIGN KEY (user_id)
REFERENCES public.user(id)
ON DELETE SET NULL;

CREATE INDEX user_id_index
ON public.employee(user_id);

CREATE INDEX emp_lastname_index
ON public.employee(last_name);

CREATE INDEX pesel_index
ON public.employee(pesel);
